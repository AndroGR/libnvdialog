Quick Start

This is a quick example of a simple NvDialog dialog window,
according to the v0.2.0 API. 


#include <stdio.h>
#include <stdlib.h>
#include <nvdialog/nvdialog.h>

int main(int argc, char** argv) {
    if (nvd_init(argv[0]) != 0) {
        puts("NvDialog failed to initialize, aborting.");
        abort();
    }

    char* title = "Hello World!";
    char* msg   = "Hello World! This is a dialog box created using NvDialog.";
    NvdDialogBox* box = nvd_dialog_box_new(title,
                                           msg,
                                           NVD_DIALOG_SIMPLE);
    
    if (!box) exit(EXIT_FAILURE);
    nvd_show_dialog(box);
    nvd_free_object(box);

    return 0;
}
